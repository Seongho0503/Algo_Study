import sys
sys.stdin = open("input.txt")
input = sys.stdin.readline

cash = int(input())
stock_price = list(map(int, input().split()))

remain_bnp = remain_timing = cash   # ì´ˆê¸° ìì‚°
stock_bnp = stock_timing = 0        # ì´ˆê¸° ì£¼ì‹ìˆ˜

up = 0
down = 0
for i, price in enumerate(stock_price):
  # BNP
  stock_bnp += remain_bnp // price
  remain_bnp %= price
  
  # TIMING
  if i > 0: # ì¸ë±ìŠ¤ ì—ëŸ¬ ë°©ì§€
    if stock_price[i] > stock_price[i-1]: # ë” í¬ë‹¤ë©´
      up += 1
      down = 0
    elif stock_price[i] < stock_price[i-1]: # ë” ì‘ë‹¤ë©´
      down += 1
      up = 0
    else:
      down = up = 0
  if up >= 3: # 3ì¼ ì—°ì† ìƒìŠ¹ì„¸ì´ë©´, íŒë‹¤
    remain_timing += price * stock_timing
    stock_timing = 0
  elif down >= 3: # 3ì¼ ì—°ì† í•˜ë½ì„¸ë©´, ì‚°ë‹¤.
    stock_timing += remain_timing // price
    remain_timing %= price

bnp = stock_price[-1] * stock_bnp + remain_bnp
timing = stock_price[-1] * stock_timing + remain_timing
if bnp > timing:
  print("BNP")
elif bnp < timing:
  print("TIMING")
else:
  print("SAMESAME")
